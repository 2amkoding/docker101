mkdir test                :    to create 'test' directory
mv test docker            :     to rename to 'docker'
touch file1.txt file2.txt :     to create files
mv file1.txt hello.txt    :    rename to hello
rm hello.txt              :  removes file
rm file*                  :  removes all files that start with 'file'
rm -r docker              :  recursively removes a directory
